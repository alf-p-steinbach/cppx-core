18  //------------------------------------------ hopefully & fail
19  //
20  // The default `fail` function's exception can be caught as a `std::runtime_error`.
21  // It's the current exception, if any, that's nested. And the nested exception
22  // /can/ just be ignored at the exception handling site.
23  //
24  // The rationale of the nesting is to not discard useful information even though
25  // it carries some overhead, because throwing should be rare, not the normal case.
26  //
27  // Typical usage patterns:
28  //
29  // 1;
30  // getline( cin, s )
31  // or fail( "foo - getline failed" );
32  //
33  // 2;
34  // const ptrdiff_t n_values = to_wide_from_utf8( &s[0], &wide_s[0] );
35  // hopefully( n_values >= 0 )
36  // or fail( "foo - to_wide_from_utf8 failed" );
37  // wide_s.resize( n_values );
38  //
39  // In future `fail` may be extended to support error codes by throwing a
40  // `std::system_error`, which is derived from `std::runtime_error` and carries an
41  // error code.

77  //------------------------------------------ Success, Failure & `>>`
78  //
79  // A unified notation that provides separation of failure /checking/ and
80  // /throwing/ of exception with possibly costly to construct arguments.
81  //
82  // Typical usage patterns, here using a Windows' function that returns `HRESULT`:
83  //
84  // auto operator>>( const HRESULT hr, Success )

86  // { return SUCCEEDED( hr ); }

91  //
92  // Since the `or` is the built-in `or` the arguments to `fail` are only evaluated
93  // in the case of failure, which means the normal case is efficient.
